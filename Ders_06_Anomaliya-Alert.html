Anomaliya aşkarlama və xəbərdarlıq

İndiyə qədər sistem məlumatı topladı və saxladı. Bu dərsdə ona «ağıl» əlavə edirik. Hər gələn ölçmə sensor_tipi cədvəlindəki normal həddlərlə müqayisə olunacaq. Temperatur 35°C-i keçsə, vibrasiya 7 mm/s-i aşsa — xeberdarliq cədvəlinə qeyd yazılacaq və real-vaxt alarm veriləcək.

01

İki növ yoxlama — fərqi anlayaq

Dərs 5-də artıq bir validasiya qatı qurduq. Bu dərsdəki anomaliya yoxlaması fərqli məqsəd daşıyır. Onları qarışdırmamaq vacibdir.

Əsas fərq
Anomal dəyər həqiqi ölçmədir — onu atmaq olmaz, əksinə saxlamaq və diqqət çəkmək lazımdır. 42°C real problemdir (mühərrik qızır?) və bazada qalmalıdır ki, sonra analiz oluna bilsin. Yalnız əlavə olaraq bir xəbərdarlıq qeydi yaranır.
02

03

Həddləri bazadan oxumaq

Həddlər sensor_tipi cədvəlində saxlanılır (Dərs 2-də seed etdik). Məsələn temperatur üçün min_hedd=-5, max_hedd=35. Anomaliya modulu əvvəlcə bu həddləri yükləyəcək.

Amma hər ölçmədə bazadan hədd oxumaq yavaş olardı (saniyədə çoxlu mesaj gəlir). Ona görə həddləri bir dəfə oxuyub yaddaşda keş (cache) saxlayırıq. Cihaz→hədd xəritəsi.

Dizayn qərarı
Həddləri xidmət başlayanda bir dəfə yükləyirik. Əgər həddlər sonradan dəyişsə (nadir hadisə), xidməti yenidən başlatmaq kifayətdir. Bu, sadəlik və sürət arasında düzgün balansdır — real sənaye sistemlərində də belə edilir.
04

Fayl 1 — anomaliya modulu avtomatik yaranır

Bu modul həddləri yükləyir, bir ölçməni yoxlayır və hədd aşılıbsa xeberdarliq cədvəlinə yazır. Xidmətdən ayrı fayl saxlayırıq ki, məntiq təmiz və test edilə bilən olsun.

"""
Anomaliya askarlama - olcmeni heddlerle muqayise edir.
Fayl: 03_src/alert/anomaliya.py
"""
import logging

log = logging.getLogger("anomaliya")

# cihaz_kod -> (min_hedd, max_hedd, vahid, ad)
_heddler = {}


def heddleri_yukle(conn):
    """sensor_tipi ve cihaz cedvellerinden heddleri yaddasa yuklyeyir."""
    global _heddler
    _heddler = {}
    with conn.cursor() as cur:
        cur.execute("""
            SELECT c.kod, t.min_hedd, t.max_hedd, t.vahid, t.ad
            FROM cihaz c
            JOIN sensor_tipi t ON c.sensor_tipi_kod = t.kod
        """)
        for kod, minh, maxh, vahid, ad in cur.fetchall():
            _heddler[kod] = {
                "min": float(minh) if minh is not None else None,
                "max": float(maxh) if maxh is not None else None,
                "vahid": vahid,
                "ad": ad,
            }
    log.info(f"Heddler yuklendi: {len(_heddler)} cihaz")
    return _heddler


def yoxla(cihaz_kod, qiymet):
    """
    Bir olcmeni yoxlayir.
    Qaytarir: None (normal) ve ya (novu, mesaj) (anomaliya).
    """
    h = _heddler.get(cihaz_kod)
    if h is None:
        return None  # namelum cihaz - yoxlanmir

    if h["max"] is not None and qiymet > h["max"]:
        mesaj = (f"{h['ad']}: {qiymet} {h['vahid']} "
                 f"(max hedd {h['max']} {h['vahid']} asildi)")
        return ("yuxari_hedd", mesaj)

    if h["min"] is not None and qiymet < h["min"]:
        mesaj = (f"{h['ad']}: {qiymet} {h['vahid']} "
                 f"(min hedd {h['min']} {h['vahid']} altinda)")
        return ("asagi_hedd", mesaj)

    return None


def xeberdarliq_yaz(conn, cihaz_kod, olcme_vaxti, qiymet, novu, mesaj):
    """xeberdarliq cedveline qeyd yazir."""
    with conn.cursor() as cur:
        cur.execute("""
            INSERT INTO xeberdarliq
                (cihaz_kod, olcme_vaxti, qiymet, novu, mesaj)
            VALUES (%s, %s, %s, %s, %s)
        """, (cihaz_kod, olcme_vaxti, qiymet, novu, mesaj))
EOF

"""
FastAPI toplama xidmeti - MQTT dinleyir, validasiya edir, bazaya yazir,
anomaliyalari askarlayir (Ders 6).
Fayl: 03_src/toplama/xidmet.py
Ishe salmaq:
  PYTHONPATH=.:../alert uvicorn xidmet:app --host 0.0.0.0 --port 8020
"""
import os
import sys
import json
import logging
from contextlib import asynccontextmanager

import psycopg
import paho.mqtt.client as mqtt
from paho.mqtt.enums import CallbackAPIVersion
from fastapi import FastAPI
from pydantic import ValidationError

# alert qovlugunu yola elave et
sys.path.insert(0, os.path.join(os.path.dirname(__file__), "..", "alert"))

from db_komekci import edge_dsn
from validasiya import OlcmeMesaj
from mqtt_komekci import MQTT_HOST, MQTT_PORT, TOPIC_BASE
import anomaliya

logging.basicConfig(level=logging.INFO,
                    format="%(asctime)s  %(levelname)s  %(message)s")
log = logging.getLogger("toplama")

stat = {"qebul": 0, "yazildi": 0, "redd": 0, "xeberdarliq": 0}

_db = None


def db_qosul():
    global _db
    if _db is None or _db.closed:
        _db = psycopg.connect(edge_dsn(), autocommit=True)
    return _db


def olcme_yaz(m: OlcmeMesaj):
    conn = db_qosul()
    with conn.cursor() as cur:
        cur.execute(
            """INSERT INTO olcme (cihaz_kod, olcme_vaxti, qiymet)
               VALUES (%s, %s, %s)""",
            (m.cihaz_kod, m.olcme_vaxti, m.qiymet),
        )
    # --- ANOMALIYA YOXLAMASI (Ders 6) ---
    netice = anomaliya.yoxla(m.cihaz_kod, m.qiymet)
    if netice is not None:
        novu, mesaj = netice
        anomaliya.xeberdarliq_yaz(conn, m.cihaz_kod, m.olcme_vaxti,
                                  m.qiymet, novu, mesaj)
        stat["xeberdarliq"] += 1
        log.warning(f"ALARM [{novu}] {mesaj}")


def on_connect(client, userdata, flags, reason_code, properties):
    if reason_code == 0:
        topic = f"{TOPIC_BASE}/#"
        client.subscribe(topic, qos=0)
        log.info(f"Broker-e qosuldu, abune: {topic}")
    else:
        log.error(f"Broker qosulmasi alinmadi: {reason_code}")


def on_message(client, userdata, msg):
    stat["qebul"] += 1
    try:
        data = json.loads(msg.payload.decode())
        m = OlcmeMesaj(**data)
        olcme_yaz(m)
        stat["yazildi"] += 1
    except (ValidationError, ValueError) as e:
        stat["redd"] += 1
        log.warning(f"REDD ({msg.topic}): {e}")
    except Exception as e:
        stat["redd"] += 1
        log.error(f"XETA ({msg.topic}): {e}")


mqtt_client = mqtt.Client(
    callback_api_version=CallbackAPIVersion.VERSION2,
    client_id="zarat-toplama",
)
mqtt_client.on_connect = on_connect
mqtt_client.on_message = on_message


@asynccontextmanager
async def lifespan(app: FastAPI):
    # baslanğic: heddleri yukle, broker-e qosul
    conn = db_qosul()
    anomaliya.heddleri_yukle(conn)
    mqtt_client.connect(MQTT_HOST, MQTT_PORT, keepalive=60)
    mqtt_client.loop_start()
    log.info("Toplama xidmeti basladi (anomaliya aktiv).")
    yield
    mqtt_client.loop_stop()
    mqtt_client.disconnect()
    log.info("Toplama xidmeti dayandi.")


app = FastAPI(title="Zarat Faza 2 - Toplama Xidmeti", lifespan=lifespan)


@app.get("/")
def kok():
    return {"xidmet": "Zarat toplama", "veziyyet": "isleyir"}


@app.get("/saglamliq")
def saglamliq():
    return {"status": "ok", "statistika": stat}


@app.get("/statistika")
def statistika():
    return stat


@app.get("/xeberdarliqlar")
def xeberdarliqlar():
    """Son 20 xeberdarligi qaytarir."""
    conn = db_qosul()
    with conn.cursor() as cur:
        cur.execute("""
            SELECT cihaz_kod, novu, mesaj, olcme_vaxti
            FROM xeberdarliq
            ORDER BY yaradilma DESC LIMIT 20
        """)
        setirler = cur.fetchall()
    return [
        {"cihaz": r[0], "novu": r[1], "mesaj": r[2], "vaxt": str(r[3])}
        for r in setirler
    ]
EOF

Bonus — süni anomaliya testi

Anomaliyanın dəqiq işlədiyini əl ilə də sınaya bilərik. Xüsusi olaraq həddi aşan bir mesaj göndərək:

terminal — süni anomaliya
# temperatur 50 derece - yuxari hedd (35) asilir
mosquitto_pub -h localhost -p 1883 \
  -t "zavod/siyezen/temperatur/S001" \
  -m '{"cihaz_kod":"S001","olcme_vaxti":"2026-07-10T05:00:00+00:00","qiymet":50}'

# ceki menfi - asagi hedd (0) altinda
mosquitto_pub -h localhost -p 1883 \
  -t "zavod/siyezen/ceki/S003" \
  -m '{"cihaz_kod":"S003","olcme_vaxti":"2026-07-10T05:00:00+00:00","qiymet":-3}'

Terminal C dərhal iki alarm göstərməlidir:

WARNING ALARM [yuxari_hedd] Temperatur: 50.0 °C (max hedd 35.0 °C asildi) WARNING ALARM [asagi_hedd] Ceki (dozator): -3.0 kg (min hedd 0.0 kg altinda)
Diqqət
Çəki -3 kg anomaliyadır (asagi_hedd), amma validasiyadan keçir (çünki -3 mənasız deyil, sadəcə həddin altındadır). Bu, iki qatın fərqini göstərir: -99999 rədd olunardı, -3 isə saxlanılır + xəbərdarlıq yaradır.
