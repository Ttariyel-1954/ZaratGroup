02 — Keş

Hədd keşinin genişləndirilməsi

Hazırda HEDLER keşi hər cihaz üçün üç dəyər saxlayır: (min, max, vahid). İndi alert parametrləri də lazımdır.

03_src/api/baza.py — hedleri_yukle() əvəzlənir
# Köhnə: HEDLER = {"S001": (min, max, vahid)}
# Yeni:  HEDLER = {"S001": {tam parametr dəsti}}

HEDLER: dict[str, dict] = {}


async def hedleri_yukle() -> int:
    """
    Cihaz hədlərini VƏ alert parametrlərini yükləyir.
    Bərpa/kritik hədləri BURADA hesablanır — hər ölçmədə yox.
    """
    sql = """
        SELECT c.kod, t.kod AS tip, t.vahid,
               t.min_hedd, t.max_hedd,
               t.ardicil_hedd, t.hysteresis_faiz, t.kritik_faiz
        FROM cihaz c
        JOIN sensor_tipi t ON c.sensor_tipi_kod = t.kod
        WHERE c.status = 'aktiv'
    """
    async with hovuz.connection() as conn:
        async with conn.cursor() as cur:
            await cur.execute(sql)
            setirler = await cur.fetchall()

    HEDLER.clear()
    for (kod, tip, vahid, mn, mx,
         ardicil, hyst_faiz, kritik_faiz) in setirler:

        mn, mx = float(mn), float(mx)
        diapazon = mx - mn
        hyst = float(hyst_faiz) / 100
        krit = float(kritik_faiz) / 100

        HEDLER[kod] = {
            "tip":   tip,
            "vahid": vahid,
            "min":   mn,
            "max":   mx,

            # Hysteresis: alert BU dəyərlərdə bağlanır
            "berpa_yuxari": mx - diapazon * hyst,   # 35 - 40*0.06 = 32.6
            "berpa_asagi":  mn + diapazon * hyst,   # -5 + 40*0.06 = -2.6

            # Kritik: bu dəyərlərdən sonra səviyyə = "kritik"
            "kritik_yuxari": mx * (1 + krit),          # 35 * 1.20 = 42.0
            "kritik_asagi":  mn * (1 - krit) if mn > 0
                               else mn * (1 + krit),

            "ardicil_hedd": int(ardicil),
        }

    return len(HEDLER)
Diqqət: kritik_asagi-da işarə tələsi min_hedd mənfi ola bilər (temperatur: −5 °C). Onda mn * (1 - 0.20) = −5 × 0.8 = −4 — bu, səhvdir: −4, −5-dən yuxarıdır, yəni kritik hədd əsl həddən daha yumşaqdır.

Düzgün: mənfi hədd üçün mn * (1 + 0.20) = −6 — daha aşağı, daha sərt. Yuxarıdakı kodda if mn > 0 yoxlaması məhz bunu edir.

Bu, kiçik detaldır, amma yoxlanmasa alert heç vaxt kritikə çevrilməzdi və heç kim səbəbini tapmazdı.
Nəticə — TEMP (S001) üçün
{
  "tip": "TEMP", "vahid": "°C",
  "min": -5.0,           "max": 35.0,
  "berpa_asagi": -2.6,  "berpa_yuxari": 32.6,   // alert BURADA bağlanır
  "kritik_asagi": -6.0, "kritik_yuxari": 42.0,  // buradan sonra KRİTİK
  "ardicil_hedd": 5                              // 5 ardıcıl aşım lazımdır
}

03 — Mühərrik

alert.py — vəziyyət maşını

03_src/api/alert.py
"""
Alert mühərriki — Dərs 06.
Vəziyyət maşını: NORMAL → GÖZLƏMƏ → AKTİV → BAĞLANMIŞ
Dörd mexanizm: ardıcıllıq, debouncing, hysteresis, səviyyələr.
"""
import logging
from dataclasses import dataclass

from .baza import HEDLER

log = logging.getLogger("zarat.alert")

# Ardıcıllıq sayğacları — YADDAŞDA saxlanır
# {"S001": {"yuxari_hedd": 2, "asagi_hedd": 0}}
SAYGAC: dict[str, dict[str, int]] = {}

ALERT_METRIKA = {"acilan": 0, "yenilenen": 0,
                 "baglanan": 0, "kritik": 0, "udulan": 0}


@dataclass
class AlertEmri:
    """Alert mühərrikinin qərarı — yazıcı bunu icra edəcək."""
    emr: str            # "ac_ve_ya_yenile" | "bagla" | "hec_ne"
    cihaz_kod: str
    novu: str or None   # "yuxari_hedd" | "asagi_hedd"
    seviyye: str = "xeberdarliq"
    qiymet: float = 0.0
    olcme_vaxti = None
    mesaj: str = ""


def sayqac_al(kod: str, novu: str) -> int:
    return SAYGAC.setdefault(kod, {}).get(novu, 0)


def sayqac_artir(kod: str, novu: str) -> int:
    d = SAYGAC.setdefault(kod, {})
    d[novu] = d.get(novu, 0) + 1
    return d[novu]


def sayqac_sifirla(kod: str, novu: str) -> None:
    SAYGAC.setdefault(kod, {})[novu] = 0


def qiymetlendir(kod: str, qiymet: float, vaxt) -> AlertEmri:
    """
    Bir ölçməni qiymətləndirir və NƏ EDİLMƏLİ olduğunu qaytarır.
    Bazaya YAZMIR — yalnız QƏRAR verir. Test etmək asandır.
    """
    h = HEDLER.get(kod)
    if h is None:
        return AlertEmri("hec_ne", kod, None)

    vahid = h["vahid"]

    # ---------- 1. YUXARI HƏDD ----------
    if qiymet > h["max"]:
        say = sayqac_artir(kod, "yuxari_hedd")

        # ARDICILLIQ: hələ kifayət deyil → SUS
        if say < h["ardicil_hedd"]:
            ALERT_METRIKA["udulan"] += 1
            log.debug("%s: aşım %d/%d — hələ alert yox",
                      kod, say, h["ardicil_hedd"])
            return AlertEmri("hec_ne", kod, "yuxari_hedd")

        # SƏVİYYƏ: kritikdirmi?
        seviyye = "kritik" if qiymet >= h["kritik_yuxari"] else "xeberdarliq"

        return AlertEmri(
            emr="ac_ve_ya_yenile", cihaz_kod=kod, novu="yuxari_hedd",
            seviyye=seviyye, qiymet=qiymet, olcme_vaxti=vaxt,
            mesaj=f"{kod}: {qiymet} {vahid} > maks {h['max']} {vahid}",
        )

    # ---------- 2. AŞAĞI HƏDD ----------
    if qiymet < h["min"]:
        say = sayqac_artir(kod, "asagi_hedd")

        if say < h["ardicil_hedd"]:
            ALERT_METRIKA["udulan"] += 1
            return AlertEmri("hec_ne", kod, "asagi_hedd")

        seviyye = "kritik" if qiymet <= h["kritik_asagi"] else "xeberdarliq"

        return AlertEmri(
            emr="ac_ve_ya_yenile", cihaz_kod=kod, novu="asagi_hedd",
            seviyye=seviyye, qiymet=qiymet, olcme_vaxti=vaxt,
            mesaj=f"{kod}: {qiymet} {vahid} < min {h['min']} {vahid}",
        )

    # ---------- 3. HYSTERESIS ZOLAĞI ----------
    # Dəyər hədlər arasındadır, AMMA bərpa həddinə çatmayıb.
    # → alert AÇIQ qalmalıdır. Heç nə etmirik, sayğacı da SIFIRLAMIRIQ.
    if qiymet > h["berpa_yuxari"] or qiymet < h["berpa_asagi"]:
        return AlertEmri("hec_ne", kod, None)

    # ---------- 4. TAM NORMAL → ALERTLƏRİ BAĞLA ----------
    sayqac_sifirla(kod, "yuxari_hedd")
    sayqac_sifirla(kod, "asagi_hedd")

    return AlertEmri(
        emr="bagla", cihaz_kod=kod, novu=None,
        qiymet=qiymet, olcme_vaxti=vaxt,
    )

Kritik incəliklərin izahı

3-cü blok — hysteresis zolağı
Bu, funksiyanın ən vacib hissəsidir. Dəyər həddi keçib (35-dən aşağı), amma bərpa həddinə (32.6) hələ düşməyib.
Nə edirik: "hec_ne" qaytarırıq — alert açıq qalır.
Sayğacı niyə SIFIRLAMIRIQ: əgər sıfırlasaq və dəyər yenidən 35-i keçsə, ardıcıllıq sayğacı yenidən 0-dan başlayardı — və 5 ölçmə gözləmək lazım gələrdi. Halbuki problem heç bitməmişdi. Zolaqda sayğac toxunulmaz qalır.
4-cü blok — sayğacların sıfırlanması
Yalnız tam normal zonada sıfırlanır (bərpa hədləri arasında). Bu, "problem həqiqətən bitdi" siqnalıdır.
qiymetlendir() bazaya YAZMIR
Şüurlu dizayn. Funksiya yalnız qərar verir və AlertEmri qaytarır. Bazaya yazmaq yazıcının işidir.
Faydası: bu funksiyanı bazasız test etmək olar — sadəcə çağırıб nəticəyə baxırsınız. Yan təsiri olmayan (pure) funksiya həmişə test-dostudur.
SAYGAC — yaddaşda
Ardıcıllıq sayğacları bazada yox, yaddaşdadır.
Niyə: hər ölçmədə bazaya yazsaq, saniyədə onlarla lüzumsuz UPDATE olardı.
Kompromis: servis restart olsa, sayğaclar sıfırlanır — yəni alert bir neçə ölçmə gec açılar. Qəbul edilə bilən itki. Aktiv alertlər isə bazadadır və itmir.

04 — Upsert

Bir SQL: yoxdursa yarat, varsa yenilə

Debouncing-i kodda belə yaza bilərdik:

SƏHV yanaşma — yarış şəraiti (race condition)
# 1) Aktiv alert var?
aktiv = await aktiv_alert_tap(kod, novu)

if aktiv:
    await alerti_yenile(aktiv.id)     # 2a) yenilə
else:
    await alert_yarat(kod, novu)     # 2b) yarat

# PROBLEM: 1 və 2 arasında BAŞQA bir proses alert yarada bilər
#          → iki alert → indeks xətası → dəstə itir

Düzgün yanaşma: upsert — bir atomik SQL.

03_src/api/alert.py — davamı
SQL_UPSERT = """
INSERT INTO xeberdarliq (
    cihaz_kod, olcme_vaxti, qiymet, novu, mesaj,
    seviyye, acilma_vaxti, tetik_sayi,
    ilk_qiymet, son_qiymet, pik_qiymet, hell_olundu
)
VALUES (%(kod)s, %(vaxt)s, %(qiymet)s, %(novu)s, %(mesaj)s,
        %(seviyye)s, %(vaxt)s, 1,
        %(qiymet)s, %(qiymet)s, %(qiymet)s, FALSE)

ON CONFLICT (cihaz_kod, novu) WHERE hell_olundu = FALSE
DO UPDATE SET
    tetik_sayi = xeberdarliq.tetik_sayi + 1,
    son_qiymet = EXCLUDED.son_qiymet,
    qiymet     = EXCLUDED.qiymet,
    mesaj      = EXCLUDED.mesaj,

    -- Pik: yuxarı hədd üçün MAKS, aşağı hədd üçün MİN
    pik_qiymet = CASE
        WHEN xeberdarliq.novu = 'yuxari_hedd'
            THEN GREATEST(xeberdarliq.pik_qiymet, EXCLUDED.pik_qiymet)
        ELSE LEAST(xeberdarliq.pik_qiymet, EXCLUDED.pik_qiymet)
    END,

    -- Səviyyə yalnız YÜKSƏLƏ bilər, enməz
    seviyye = CASE
        WHEN EXCLUDED.seviyye = 'kritik' THEN 'kritik'
        ELSE xeberdarliq.seviyye
    END

RETURNING id, (xmax = 0) AS yeni_yaradildi, seviyye, tetik_sayi
"""


async def alerti_yaz(conn, emr: AlertEmri) -> dict | None:
    """Upsert icra edir. Nəticəni qaytarır — bildiriş üçün lazımdır."""
    async with conn.cursor() as cur:
        await cur.execute(SQL_UPSERT, {
            "kod":     emr.cihaz_kod,
            "vaxt":    emr.olcme_vaxti,
            "qiymet":  emr.qiymet,
            "novu":    emr.novu,
            "mesaj":   emr.mesaj,
            "seviyye": emr.seviyye,
        })
        setir = await cur.fetchone()

    if setir is None:
        return None

    alert_id, yeni, seviyye, tetik = setir

    if yeni:
        ALERT_METRIKA["acilan"] += 1
        log.warning("ALERT AÇILDI [%s] %s — %s",
                    seviyye.upper(), emr.cihaz_kod, emr.mesaj)
    else:
        ALERT_METRIKA["yenilenen"] += 1
        log.debug("Alert yeniləndi: %s (tətik %d)", emr.cihaz_kod, tetik)

    if seviyye == "kritik":
        ALERT_METRIKA["kritik"] += 1

    return {"id": alert_id, "yeni": yeni,
            "seviyye": seviyye, "tetik_sayi": tetik}
ON CONFLICT (...) WHERE hell_olundu = FALSE
Bu WHERE Hissə A-dakı qismən indeksə istinad edir. Onsuz PostgreSQL hansı indeksi nəzərdə tutduğunuzu bilməz və xəta verər. Yəni bazadakı indeks ilə kodakı upsert bir-birinə bağlıdır — biri olmasa, digəri işləməz.
xmax = 0
PostgreSQL hiyləsi. Bu, "sətir yeni yaradıldımı, yoxsa yeniləndimi?" sualına cavab verir. xmax = 0 → yeni sətir.
Niyə lazımdır: bildiriş yalnız yeni açılan alert üçün göndərilməlidir. Hər yeniləmədə e-poçt göndərsək — yenidən sel yaranar, sadəcə poçt qutusunda.
pik_qiymet — CASE ilə
Yuxarı hədd alerti üçün pik = ən böyük dəyər (GREATEST). Aşağı hədd üçün pik = ən kiçik (LEAST) — çünki orada "ən pis" daha aşağı deməkdir.
seviyye — yalnız yüksəlir
Alert kritik olubsa, sonrakı ölçmə 36 °C olsa belə (kritik həddin altında), səviyyə kritik qalır.
Niyə: problem kritik həddə çatıb — bu fakt itməməlidir. Operator alerti bağlayanda tam hekayəni görməlidir.

05 — Bağlanma

Alertin bağlanması

03_src/api/alert.py — davamı
SQL_BAGLA = """
UPDATE xeberdarliq
SET hell_olundu    = TRUE,
    baglanma_vaxti = %(vaxt)s,
    son_qiymet     = %(qiymet)s
WHERE cihaz_kod = %(kod)s
  AND hell_olundu = FALSE
  AND novu IN ('yuxari_hedd', 'asagi_hedd')
RETURNING id, novu, seviyye, tetik_sayi,
          (baglanma_vaxti - acilma_vaxti) AS muddet
"""


async def alertleri_bagla(conn, emr: AlertEmri) -> list:
    """
    Cihazın hədd alertlərini bağlayır (dəyər normala qayıdıb).
    cihaz_susur alertinə TOXUNMUR — onu nezaretci idarə edir.
    """
    async with conn.cursor() as cur:
        await cur.execute(SQL_BAGLA, {
            "kod":    emr.cihaz_kod,
            "vaxt":   emr.olcme_vaxti,
            "qiymet": emr.qiymet,
        })
        baglananlar = await cur.fetchall()

    for alert_id, novu, seviyye, tetik, muddet in baglananlar:
        ALERT_METRIKA["baglanan"] += 1
        log.info("ALERT BAĞLANDI %s [%s] — %s ərzində, %d tətik",
                 emr.cihaz_kod, novu, muddet, tetik)

    return baglananlar
Niyə novu IN ('yuxari_hedd', 'asagi_hedd')? Cihaz normal dəyər göndərirsə, onun hədd alertləri bağlanmalıdır. Amma cihaz_susur alerti ayrı məntiqlə idarə olunur — o, mesaj gəldiyi anda bağlanır (dəyər normal olmasa da).

Bu şərt olmasa: cihaz susub (alert var), sonra anomal dəyər göndərib — susma alerti bağlanmazdı, çünki dəyər normal deyil. Halbuki cihaz danışır! Məntiq səhv olardı.

Susma alertini bölmə 07-də ayrıca bağlayacağıq.

06 — İnteqrasiya

yazici.py-nin yenidən qurulması

İki if sətri gedir, alert mühərriki gəlir.

03_src/api/yazici.py — deste_yaz() əvəzlənir
from . import alert as alert_modul
from .bildiris import bildiris_gonder

SQL_OLCME = """
    INSERT INTO olcme (cihaz_kod, olcme_vaxti, qiymet, keyfiyyet)
    VALUES (%s, %s, %s, %s)
    ON CONFLICT (cihaz_kod, olcme_vaxti) DO NOTHING
"""


async def deste_yaz(deste: list) -> None:
    """Dəstəni bir tranzaksiyada yazır + alert mühərrikini işə salır."""
    olcme_setirleri = []
    emrler = []

    # ---- 1. Qiymətləndir (bazasız, sürətli) ----
    for o in deste:
        h = HEDLER.get(o.cihaz_kod)
        if h is None:
            log.warning("Naməlum cihaz: %s", o.cihaz_kod)
            YAZICI_METRIKA["xeta"] += 1
            continue

        keyfiyyet = 1 if h["min"] <= o.qiymet <= h["max"] else 0
        olcme_setirleri.append(
            (o.cihaz_kod, o.olcme_vaxti, o.qiymet, keyfiyyet)
        )
        if keyfiyyet == 0:
            YAZICI_METRIKA["anomal"] += 1

        # ★ Alert mühərriki — QƏRAR verir, yazmır ★
        emrler.append(alert_modul.qiymetlendir(
            o.cihaz_kod, o.qiymet, o.olcme_vaxti
        ))

    if not olcme_setirleri:
        return

    yeni_kritikler = []

    # ---- 2. Bir tranzaksiyada hər şeyi yaz ----
    try:
        async with hovuz.connection() as conn:
            async with conn.cursor() as cur:
                await cur.executemany(SQL_OLCME, olcme_setirleri)

            # Alert əmrlərini icra et
            for emr in emrler:
                if emr.emr == "ac_ve_ya_yenile":
                    netice = await alert_modul.alerti_yaz(conn, emr)
                    # Bildiriş YALNIZ yeni açılan kritik alert üçün
                    if netice and netice["yeni"] and netice["seviyye"] == "kritik":
                        yeni_kritikler.append(emr)

                elif emr.emr == "bagla":
                    await alert_modul.alertleri_bagla(conn, emr)

        YAZICI_METRIKA["yazilan"] += len(olcme_setirleri)
        YAZICI_METRIKA["batch"] += 1

    except psycopg.Error as e:
        YAZICI_METRIKA["xeta"] += len(olcme_setirleri)
        log.error("Baza xətası, dəstə (%d) itdi: %s",
                  len(olcme_setirleri), e)
        return

    # ---- 3. Bildiriş TRANZAKSİYADAN SONRA ----
    for emr in yeni_kritikler:
        await bildiris_gonder(
            baslik=f"KRİTİK: {emr.cihaz_kod}",
            metn=emr.mesaj,
        )
Niyə bildiriş tranzaksiyadan SONRA? İki səbəb:

① E-poçt göndərmək şəbəkə əməliyyatıdır — saniyələr çəkə bilər. Tranzaksiyanın içində etsək, baza bağlantısı və kilidlər bütün o müddət tutulu qalar. Digər yazmalar gözləyər.

② Tranzaksiya ROLLBACK olarsa, e-poçt artıq getmiş olar — operator mövcud olmayan alert haqqında bildiriş alar. Utancverici və çaşdırıcı.

Qayda: tranzaksiyanın içində yalnız baza əməliyyatı. Xarici dünya ilə ünsiyyət — commit-dən sonra.

07 — Susan cihaz

cihaz_susur: LWT-nin vədi

Dərs 04-A-da demişdik: "Susan sensor xarab sensordan təhlükəlidir. Onu LWT ilə tutacağıq." İndi o vədin vaxtıdır.

Niyə susma ən təhlükəli haldır Sensor səhv dəyər göndərsə — qrafikdə görünür, alert yaranır, operator reaksiya verir.

Sensor heç nə göndərməsə — heç nə baş vermir. Qrafik sadəcə düz xətt kimi davam edir (son dəyərdə donur). Operator "hər şey normaldır" düşünür.

Halbuki silo yana bilər, yataq dağıla bilər — və heç kim bilməz, çünki xəbər verən susub.

İki aşkarlama üsulu — hər ikisi lazımdır

ÜsulNecə işləyirNə tuturNəyi tuta bilmir
① LWT Broker cihazın qoşulmasının kəsildiyini görüb "offline" elan edir Cihaz söndü / şəbəkəsi kəsildi — 45 saniyəyə aşkarlanır Cihaz qoşuludur, amma sensor xarabdır və mesaj göndərmir
② Vaxt nəzarəti "Son ölçmə N dəqiqədən köhnədirsə → susub" Hər cür susma — o cümlədən "qoşulu, amma lal" halı Gec aşkarlayır (N dəqiqə)
Niyə ikisi birlikdə LWT sürətlidir (45 san.), amma yalnız qoşulma kəsilməsini tutur. Vaxt nəzarəti yavaşdır, amma hər şeyi tutur.

Real sistemdə hər ikisi olmalıdır — biri sürət, digəri əhatə verir.

"""
Nəzarətçi — susan cihazları tutur.
Fon tapşırığı: hər 60 saniyədə bir bütün cihazları yoxlayır.
"""
import asyncio
import logging
from datetime import datetime, timezone

from .baza import hovuz, HEDLER
from .bildiris import bildiris_gonder

log = logging.getLogger("zarat.nezaretci")

YOXLAMA_INTERVAL = 60     # saniyə
SUSMA_HEDD_DEQ   = 5      # 5 dəqiqədir mesaj yoxdursa → susub

NEZARET_METRIKA = {"susan": 0, "berpa": 0, "yoxlama": 0}


# Susan cihazları tapır və alert açır
SQL_SUSANLAR = """
WITH son_olcme AS (
    SELECT c.kod,
           max(o.olcme_vaxti) AS son_vaxt
    FROM cihaz c
    LEFT JOIN olcme o ON o.cihaz_kod = c.kod
    WHERE c.status = 'aktiv'
    GROUP BY c.kod
)
SELECT kod, son_vaxt,
       now() - COALESCE(son_vaxt, now() - interval '999 days') AS susma_muddeti
FROM son_olcme
WHERE son_vaxt IS NULL
   OR son_vaxt < now() - (%(hedd)s || ' minutes')::interval
"""

SQL_SUSMA_ALERT = """
INSERT INTO xeberdarliq (
    cihaz_kod, olcme_vaxti, qiymet, novu, mesaj,
    seviyye, acilma_vaxti, tetik_sayi,
    ilk_qiymet, son_qiymet, pik_qiymet, hell_olundu
)
VALUES (%(kod)s, now(), 0, 'cihaz_susur', %(mesaj)s,
        'kritik', now(), 1, 0, 0, 0, FALSE)
ON CONFLICT (cihaz_kod, novu) WHERE hell_olundu = FALSE
DO UPDATE SET
    tetik_sayi = xeberdarliq.tetik_sayi + 1,
    mesaj      = EXCLUDED.mesaj
RETURNING id, (xmax = 0) AS yeni
"""

# Cihaz yenidən danışırsa → susma alertini bağla
SQL_SUSMA_BAGLA = """
UPDATE xeberdarliq
SET hell_olundu    = TRUE,
    baglanma_vaxti = now()
WHERE novu = 'cihaz_susur'
  AND hell_olundu = FALSE
  AND cihaz_kod IN (
      SELECT DISTINCT cihaz_kod FROM olcme
      WHERE olcme_vaxti > now() - (%(hedd)s || ' minutes')::interval
  )
RETURNING cihaz_kod, (baglanma_vaxti - acilma_vaxti) AS muddet
"""


async def bir_yoxlama():
    """Bir dövr: susanları tap, bərpa olunanları bağla."""
    NEZARET_METRIKA["yoxlama"] += 1
    yeni_susanlar = []

    async with hovuz.connection() as conn:
        # --- 1) Bərpa olunanları bağla (ƏVVƏLCƏ bu!) ---
        async with conn.cursor() as cur:
            await cur.execute(SQL_SUSMA_BAGLA, {"hedd": SUSMA_HEDD_DEQ})
            for kod, muddet in await cur.fetchall():
                NEZARET_METRIKA["berpa"] += 1
                log.info("CİHAZ BƏRPA OLUNDU: %s (%s susmuşdu)", kod, muddet)

        # --- 2) Susanları tap ---
        async with conn.cursor() as cur:
            await cur.execute(SQL_SUSANLAR, {"hedd": SUSMA_HEDD_DEQ})
            susanlar = await cur.fetchall()

        # --- 3) Hər susan üçün alert ---
        for kod, son_vaxt, muddet in susanlar:
            mesaj = (f"{kod}: {SUSMA_HEDD_DEQ} dəqiqədir mesaj yoxdur "
                     f"(son: {son_vaxt or 'heç vaxt'})")

            async with conn.cursor() as cur:
                await cur.execute(SQL_SUSMA_ALERT,
                                  {"kod": kod, "mesaj": mesaj})
                alert_id, yeni = await cur.fetchone()

            if yeni:
                NEZARET_METRIKA["susan"] += 1
                log.error("CİHAZ SUSDU: %s", mesaj)
                yeni_susanlar.append((kod, mesaj))

    # --- 4) Bildiriş (tranzaksiyadan SONRA) ---
    for kod, mesaj in yeni_susanlar:
        await bildiris_gonder(baslik=f"CİHAZ SUSDU: {kod}", metn=mesaj)


async def nezaretci_dovru():
    """Fon tapşırığı — ölməməlidir."""
    log.info("Nəzarətçi başladı (interval=%ds, hədd=%d dəq.)",
             YOXLAMA_INTERVAL, SUSMA_HEDD_DEQ)
    while True:
        try:
            await asyncio.sleep(YOXLAMA_INTERVAL)
            await bir_yoxlama()

        except asyncio.CancelledError:
            log.info("Nəzarətçi dayandırılır...")
            raise

        except Exception as e:
            log.exception("Nəzarətçidə xəta: %s", e)
Sıra: əvvəlcə BAĞLA, sonra TAP
Vacib detal. Əvvəlcə susanları tapıб sonra bərpa olunanları bağlasaq, cihaz eyni dövrdə həm "susan" siyahısına düşər, həm bağlanardı — və nəticə dövrün sırasından asılı olardı. Səliqəsizdir.

"""
Bildiriş qatı — kritik alertləri operatora çatdırır.
Hazırda: log + fayl. Sonra: Telegram / e-poçt.
"""
import logging
import os
from datetime import datetime, timezone
from pathlib import Path

log = logging.getLogger("zarat.bildiris")

KOK = Path(__file__).resolve().parents[2]
BILDIRIS_FAYL = KOK / "_LOG" / "bildiris.log"

# .env-dən (yoxdursa — bildiriş söndürülüb)
TELEGRAM_TOKEN = os.getenv("TELEGRAM_TOKEN", "")
TELEGRAM_CHAT  = os.getenv("TELEGRAM_CHAT_ID", "")

BILDIRIS_METRIKA = {"gonderilen": 0, "xeta": 0}


async def bildiris_gonder(baslik: str, metn: str) -> None:
    """
    Kritik hadisəni operatora çatdırır.
    XƏTA ATMAMALIDIR — bildiriş uğursuz olsa da, sistem işləməlidir.
    """
    indi = datetime.now(timezone.utc).isoformat()
    setir = f"[{indi}] {baslik} | {metn}"

    # --- 1) Həmişə: log + fayl ---
    log.critical("BİLDİRİŞ: %s — %s", baslik, metn)
    try:
        with open(BILDIRIS_FAYL, "a", encoding="utf-8") as f:
            f.write(setir + "\n")
        BILDIRIS_METRIKA["gonderilen"] += 1
    except OSError as e:
        BILDIRIS_METRIKA["xeta"] += 1
        log.error("Bildiriş faylı yazılmadı: %s", e)

    # --- 2) Telegram (konfiqurasiya varsa) ---
    if not (TELEGRAM_TOKEN and TELEGRAM_CHAT):
        return

    try:
        import httpx
        url = f"https://api.telegram.org/bot{TELEGRAM_TOKEN}/sendMessage"
        async with httpx.AsyncClient(timeout=10.0) as c:
            await c.post(url, json={
                "chat_id": TELEGRAM_CHAT,
                "text": f"🔴 {baslik}\n\n{metn}",
            })
        log.info("Telegram bildirişi göndərildi")

    except Exception as e:
        # Bildiriş uğursuz oldu — AMMA SİSTEM DAYANMIR
        BILDIRIS_METRIKA["xeta"] += 1
        log.error("Telegram bildirişi göndərilmədi: %s", e)
