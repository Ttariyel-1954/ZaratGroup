03_src/sync/baza.py
"""
Iki ayri baglanti hovuzu.
Edge - HEMISE elcatan olmalidir.
Merkez - kesile biler, ve bu NORMALDIR.
"""
import logging

from psycopg_pool import AsyncConnectionPool

from .konfiq import EDGE_DSN, MERKEZ_DSN

log = logging.getLogger("sync.baza")

# Edge hovuzu - kicik, cunki yalniz sync istifade edir
edge_hovuz = AsyncConnectionPool(
    conninfo=EDGE_DSN,
    min_size=1, max_size=3,
    open=False,
)

# Merkez hovuzu - TIMEOUT VACIBDIR!
# Sebeke kesilse, sonsuz gozlemek olmaz.
merkez_hovuz = AsyncConnectionPool(
    conninfo=MERKEZ_DSN,
    min_size=0,          # ← 0! Baglanti lazim olanda yaradilir
    max_size=3,
    timeout=10.0,      # ← 10 saniyeden cox gozleme
    open=False,
)


async def ac():
    await edge_hovuz.open()
    await edge_hovuz.wait()
    log.info("Edge hovuzu aciq")

    # Merkez hovuzunu ACIRIQ, amma WAIT ETMIRIK!
    # Cunki merkez elcatmaz ola biler - ve bu, servisi
    # baslamaqdan SAXLAMAMALIDIR.
    await merkez_hovuz.open()
    log.info("Merkez hovuzu aciq (elcatanliq yoxlanmadi)")


async def bagla():
    await edge_hovuz.close()
    await merkez_hovuz.close()


async def merkez_elcatandir() -> bool:
    """Merkez cavab verirmi? Istisna ATMIR."""
    try:
        async with merkez_hovuz.connection(timeout=5.0) as conn:
            async with conn.cursor() as cur:
                await cur.execute("SELECT 1")
        return True
    except Exception:
        return False

Kritik detal. min_size=1 olsa, hovuz açılanda dərhal qoşulmağa çalışar. Mərkəz əlçatmazsa — sync servisi ümumiyyətlə qalxmaz.
min_size=0 ilə: bağlantı yalnız lazım olanda yaradılır. Mərkəz sonra qalxsa, sync onu tutacaq.

Eyni səbəb. wait() "bütün bağlantılar hazır olana qədər gözlə" deməkdir. Mərkəz sönübsə, bu, əbədi gözləmədir.
timeout=10.0
Şəbəkə "yarım ölü" ola bilər: paket gedir, cavab gəlmir. Timeout olmasa, sync dondu qalar — nə xəta verər, nə işləyər. Ən pis hal.

03 — Backoff

Eksponensial gözləmə

Şəbəkə kəsildi. Sync nə etməlidir?

Sadəlövh: hər saniyə yenidən cəhd et Şəbəkə 3 saat kəsilibsə — 10 800 uğursuz cəhd.

Nəticə: log faylı şişir, CPU boşuna yanır, mərkəz bərpa olanda onlarla client eyni anda hücum edir (thundering herd) və onu yenidən yıxır.
03_src/sync/backoff.py
"""
Eksponensial backoff + jitter.
Ugursuzluqda gozleme muddeti IKI QAT artir, tavana catir.
"""
import random


class Backoff:
    def __init__(self, bas=1.0, tavan=300.0, emsal=2.0):
        self.bas    = bas       # ilk gozleme: 1 saniye
        self.tavan  = tavan     # maksimum: 5 deqiqe
        self.emsal  = emsal     # her defe 2 qat
        self.cari   = bas
        self.ugursuz_say = 0

    def ugursuz(self) -> float:
        """Ugursuzluq: gozleme muddetini artir ve qaytar."""
        self.ugursuz_say += 1
        gozleme = self.cari

        # JITTER: ±25% tesadufi deyisiklik.
        # Niye: bir nece client eyni anda cehd etmesin
        # (thundering herd problemi).
        jitter = gozleme * random.uniform(-0.25, 0.25)
        gozleme = max(0.5, gozleme + jitter)

        # Novbeti defe ucun iki qat artir (tavana qeder)
        self.cari = min(self.tavan, self.cari * self.emsal)

        return gozleme

    def ugurlu(self) -> None:
        """Ugur: her seyi sifirla."""
        self.cari = self.bas
        self.ugursuz_say = 0

    @property
    def problemdedir(self) -> bool:
        return self.ugursuz_say > 0

1~1 san.1 san.

Jitter niyə lazımdır Təsəvvür edin: üç zavod var, hər birində sync işçisi. Mərkəz 10 dəqiqə sönüb.

Jitter olmasa: hər üçü eyni cədvəllə cəhd edir — 1s, 2s, 4s... Mərkəz qalxan an üçü də eyni anda hücum edir. Mərkəz yenidən yıxılır.

Jitter ilə: cəhdlər təsadüfi yayılır. Yük hamarlanır.

Bu, kiçik detaldır — amma böyük sistemlərdə çökmə ilə sağ qalma arasındakı fərqdir.

03_src/sync/olcme_sync.py
"""Olcme sinxronizasiyasi - deyismez faktlar, DO NOTHING."""
import logging
import time

from .baza import edge_hovuz, merkez_hovuz
from .konfiq import ZAVOD_KOD, SYNC_BATCH_OLCME

log = logging.getLogger("sync.olcme")

OLCME_METRIKA = {"gonderilen": 0, "yeni": 0,
                 "dublikat": 0, "dovr": 0}

# KOHNEDEN yeniye - FIFO. Sebeb: Hisse A, bolme 10.
SQL_SEC = """
    SELECT id, cihaz_kod, olcme_vaxti, qiymet, keyfiyyet
    FROM olcme
    WHERE sync_status = 0
    ORDER BY olcme_vaxti
    LIMIT %s
"""

# IDEMPOTENT: tekrar gonderilse, sessizce atilir.
SQL_MERKEZ = """
    INSERT INTO zavod.olcme
        (zavod_kod, cihaz_kod, olcme_vaxti, qiymet, keyfiyyet)
    VALUES (%s, %s, %s, %s, %s)
    ON CONFLICT (zavod_kod, cihaz_kod, olcme_vaxti) DO NOTHING
"""

SQL_BAYRAQ = """
    UPDATE olcme
    SET sync_status = 1
    WHERE id = ANY(%s)
"""

SQL_JURNAL = """
    INSERT INTO zavod.sync_jurnal
        (zavod_kod, cedvel, setir_sayi, yeni_setir, muddet_ms)
    VALUES (%s, 'olcme', %s, %s, %s)
"""


async def olcme_gonder() -> int:
    """
    Bir dovr: gonderilmemis olcmeleri merkeze apar.
    Qaytarir: gonderilen setir sayi (0 = gonderilecek yox idi).
    ISTISNA ATIR - ana dovr onu tutub backoff isledecek.
    """
    baslangic = time.monotonic()

    # ---- 1. Edge-den oxu ----
    async with edge_hovuz.connection() as conn:
        async with conn.cursor() as cur:
            await cur.execute(SQL_SEC, (SYNC_BATCH_OLCME,))
            setirler = await cur.fetchall()

    if not setirler:
        return 0

    idler = [r[0] for r in setirler]
    merkez_setirler = [
        (ZAVOD_KOD, r[1], r[2], r[3], r[4])
        for r in setirler
    ]

    # ---- 2. MERKEZE yaz (istisna ata biler!) ----
    async with merkez_hovuz.connection() as mconn:
        async with mconn.cursor() as mcur:
            await mcur.executemany(SQL_MERKEZ, merkez_setirler)
            yeni_say = mcur.rowcount if mcur.rowcount >= 0 else len(setirler)

            muddet_ms = int((time.monotonic() - baslangic) * 1000)
            await mcur.execute(SQL_JURNAL, (
                ZAVOD_KOD, len(setirler), yeni_say, muddet_ms))
        # commit avtomatik - context menecer cixanda

    # ---- 3. YALNIZ INDI bayragi deyis ----
    # Merkez commit oldu. Indi bayragi deyismek TEHLUKESIZDIR.
    # Eger bu addim ugursuz olsa - setirler TEKRAR gonderilecek,
    # ve ON CONFLICT onlari sessizce atacaq. Zerersizdir.
    async with edge_hovuz.connection() as conn:
        async with conn.cursor() as cur:
            await cur.execute(SQL_BAYRAQ, (idler,))

    dublikat = len(setirler) - yeni_say
    OLCME_METRIKA["gonderilen"] += len(setirler)
    OLCME_METRIKA["yeni"] += yeni_say
    OLCME_METRIKA["dublikat"] += max(0, dublikat)
    OLCME_METRIKA["dovr"] += 1

    log.info("Olcme: %d gonderildi (%d yeni, %d dublikat) - %d ms",
             len(setirler), yeni_say, max(0, dublikat),
             int((time.monotonic() - baslangic) * 1000))

    return len(setirler)
Sıra — bu dərsin ən vacib üç sətri 1. Edge-dən oxu
2. Mərkəzə yaz və commit et
3. Yalnız indi bayrağı dəyiş

Bu sıra pozulsa (əvvəlcə bayraq, sonra göndərmə) — data əbədi itir. Bayraq "göndərilib" deyər, mərkəzdə isə heç nə olmaz. Və heç kim bunu tapmaz.

Düzgün sıra ilə ən pis hal: təkrar göndərmə — və ON CONFLICT onu udur.

03_src/sync/alert_sync.py
"""Alert sinxronizasiyasi - yasayan veziyyet, DO UPDATE."""
import logging
import time

from .baza import edge_hovuz, merkez_hovuz
from .konfiq import ZAVOD_KOD, SYNC_BATCH_ALERT, SYNC_ALERT_TEKRAR

log = logging.getLogger("sync.alert")

ALERT_METRIKA = {"gonderilen": 0, "yeni": 0,
                 "yenilenen": 0, "dovr": 0}

# KRITIKLER ONDE + "hot setir" mudafiesi (Hisse A, bolme 10)
SQL_SEC = """
    SELECT id, cihaz_kod, novu, seviyye, mesaj,
           acilma_vaxti, baglanma_vaxti, tetik_sayi,
           ilk_qiymet, son_qiymet, pik_qiymet, hell_olundu
    FROM xeberdarliq
    WHERE sync_status = 0
      AND (
          sync_vaxti IS NULL
          OR sync_vaxti < now() - (%s || ' seconds')::interval
      )
    ORDER BY
        (seviyye = 'kritik') DESC,
        acilma_vaxti
    LIMIT %s
"""

# DO UPDATE - cunki alert YASAYIR
SQL_MERKEZ = """
    INSERT INTO zavod.xeberdarliq (
        zavod_kod, edge_id, cihaz_kod, novu, seviyye, mesaj,
        acilma_vaxti, baglanma_vaxti, tetik_sayi,
        ilk_qiymet, son_qiymet, pik_qiymet, hell_olundu
    )
    VALUES (%s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s)
    ON CONFLICT (zavod_kod, edge_id) DO UPDATE SET
        seviyye        = EXCLUDED.seviyye,
        mesaj          = EXCLUDED.mesaj,
        baglanma_vaxti = EXCLUDED.baglanma_vaxti,
        tetik_sayi     = EXCLUDED.tetik_sayi,
        son_qiymet     = EXCLUDED.son_qiymet,
        pik_qiymet     = EXCLUDED.pik_qiymet,
        hell_olundu    = EXCLUDED.hell_olundu,
        qebul_vaxti    = now()
    RETURNING (xmax = 0) AS yeni
"""

SQL_BAYRAQ = """
    UPDATE xeberdarliq
    SET sync_status = 1, sync_vaxti = now()
    WHERE id = ANY(%s)
"""


async def alert_gonder() -> int:
    """Bir dovr: alertleri merkeze apar. ISTISNA ATIR."""
    baslangic = time.monotonic()

    async with edge_hovuz.connection() as conn:
        async with conn.cursor() as cur:
            await cur.execute(SQL_SEC,
                              (SYNC_ALERT_TEKRAR, SYNC_BATCH_ALERT))
            setirler = await cur.fetchall()

    if not setirler:
        return 0

    idler = [r[0] for r in setirler]
    yeni_say = 0

    # ---- Merkeze yaz ----
    async with merkez_hovuz.connection() as mconn:
        async with mconn.cursor() as mcur:
            for r in setirler:
                await mcur.execute(SQL_MERKEZ, (
                    ZAVOD_KOD, r[0], r[1], r[2], r[3], r[4],
                    r[5], r[6], r[7], r[8], r[9], r[10], r[11],
                ))
                netice = await mcur.fetchone()
                if netice and netice[0]:
                    yeni_say += 1

    # ---- YALNIZ INDI bayraq ----
    async with edge_hovuz.connection() as conn:
        async with conn.cursor() as cur:
            await cur.execute(SQL_BAYRAQ, (idler,))

    ALERT_METRIKA["gonderilen"] += len(setirler)
    ALERT_METRIKA["yeni"] += yeni_say
    ALERT_METRIKA["yenilenen"] += len(setirler) - yeni_say
    ALERT_METRIKA["dovr"] += 1

    log.info("Alert: %d gonderildi (%d yeni, %d yenilendi) - %d ms",
             len(setirler), yeni_say, len(setirler) - yeni_say,
             int((time.monotonic() - baslangic) * 1000))

    return len(setirler)
DO UPDATE — DO NOTHING deyil
Alert yenilənəndə (tetik_sayi artır, bağlanır) mərkəz yeni vəziyyəti görməlidir. DO NOTHING olsa, mərkəzdə alert əbədi "açıq" qalardı.
SQL_BAYRAQ-da sync_vaxti = now()
"Hot sətir" müdafiəsi. Növbəti dövrdə bu alert sync_vaxti < now() - 60s şərtinə görə seçilməyəcək — 60 saniyə keçənə qədər.
for döngüsü — executemany deyil
RETURNING nəticəsini hər sətir üçün ayrıca oxumaq lazımdır (yeni/yenilənən statistikası üçün). Alert sayı azdır (onlarla), ona görə performans problemi deyil. Ölçmə üçün isə executemany istifadə etdik — orada minlərlə sətir var.

03_src/sync/main.py
"""
Sync isciisi - ayrica proses.
Isledilir: python 03_src/sync/main.py
"""
import asyncio
import logging
import signal
import sys
from pathlib import Path

# 03_src-i yola elave et (reqemle basladigi ucun import olunmur)
sys.path.insert(0, str(Path(__file__).resolve().parents[1]))

from sync.konfiq import SYNC_INTERVAL, ZAVOD_KOD, VERSIYA
from sync.baza import ac, bagla, merkez_elcatandir
from sync.backoff import Backoff
from sync.olcme_sync import olcme_gonder, OLCME_METRIKA
from sync.alert_sync import alert_gonder, ALERT_METRIKA

logging.basicConfig(
    level=logging.INFO,
    format="%(asctime)s [%(levelname)-7s] %(name)-14s | %(message)s",
    datefmt="%Y-%m-%d %H:%M:%S",
)
log = logging.getLogger("sync")

dayan = asyncio.Event()


async def bir_dovr(backoff: Backoff) -> None:
    """Bir sync dovru: ALERTLER once, sonra olcmeler."""
    try:
        # ---- 1. ALERTLER ONCE (Hisse A, bolme 07) ----
        # Sebeke 3 saat kesilibse, kritik alert 2000 olcmenin
        # arxasinda gozlememelidir.
        alert_say = await alert_gonder()

        # ---- 2. Sonra olcmeler ----
        # Novbe bosalana qeder tekrar-tekrar
        olcme_say = await olcme_gonder()

        # Ugur - backoff sifirlanir
        if backoff.problemdedir:
            log.info("MERKEZ BERPA OLUNDU (%d ugursuz cehdden sonra)",
                     backoff.ugursuz_say)
        backoff.ugurlu()

        # Hele gonderilecek var? Derhal davam et.
        if olcme_say >= 1 or alert_say >= 1:
            return

    except Exception as e:
        gozleme = backoff.ugursuz()
        log.error("Sync ugursuz (#%d): %s - %.1f san. gozleyirem",
                  backoff.ugursuz_say, e, gozleme)
        try:
            await asyncio.wait_for(dayan.wait(), timeout=gozleme)
        except asyncio.TimeoutError:
            pass
        return

    # Gonderilecek yox idi - normal interval gozle
    try:
        await asyncio.wait_for(dayan.wait(), timeout=SYNC_INTERVAL)
    except asyncio.TimeoutError:
        pass


async def esas():
    log.info("Sync isciisi baslayir - zavod=%s, versiya=%s",
             ZAVOD_KOD, VERSIYA)
    await ac()

    if await merkez_elcatandir():
        log.info("Merkez elcatandir")
    else:
        log.warning("Merkez ELCATMAZDIR - gozleyecek, data itmeyecek")

    backoff = Backoff(bas=1.0, tavan=300.0)

    while not dayan.is_set():
        await bir_dovr(backoff)

    log.info("Sonme... Metrikalar:")
    log.info("  Olcme: %s", OLCME_METRIKA)
    log.info("  Alert: %s", ALERT_METRIKA)
    await bagla()
    log.info("Sync isciisi sondu.")


def signal_tut():
    log.info("Dayandirma siqnali alindi...")
    dayan.set()


if __name__ == "__main__":
    loop = asyncio.new_event_loop()
    asyncio.set_event_loop(loop)

    for sig in (signal.SIGINT, signal.SIGTERM):
        loop.add_signal_handler(sig, signal_tut)

    try:
        loop.run_until_complete(esas())
    finally:
        loop.close()
Növbə dolu olanda dərhal davam etmək Diqqət edin: dövrün sonunda if olcme_say >= 1: return var. Yəni göndəriləcək data qalıbsa, 10 saniyə gözləmirik — dərhal növbəti dövrə keçirik.

Niyə: şəbəkə 3 saat kəsilib, 2000 sətir yığılıб. Hər dövrdə 500 göndəririk. Əgər hər dövr arasında 10 saniyə gözləsək — bərpa 40 saniyə çəkərdi. Gözləmədən — 4 saniyə.

Növbə boşalan kimi normal 10 saniyəlik ritmə qayıdırıq.

12 — Yekun

Sistem tamamlandı

  • Outbox pattern — yazma və göndərmə tam ayrıldı. Şəbəkə kəsilsə, yazıcı xəbər tutmur.
  • Eksponensial backoff + jitter — kəsilmə zamanı ağıllı gözləmə, bərpa olanda hücum yox.
  • At-least-once + idempotentlik — cəsarətlə təkrar göndər, dublikatı baza tutur.
  • Alert prioriteti — kritik məlumat 2000 ölçmənin arxasında gözləmir.
  • İki maşınlı qurulum — pg_hba.conf, məhdud icazəli istifadəçi, ən az imtiyaz.
  • Xaos testi sübut etdi: şəbəkə kəsildi, bərpa oldu — data itmədi.

Yeddi dərsin yolu

DərsSistem nə qazandı
01–02Baza sxemi — data hara yazılacaq
03İlk ölçmələr — sistem data toplayır
04MQTT — sensorlar real protokolla danışır
05Validasiya + idempotentlik — data təmizdir
06Alert vəziyyət maşını — sistem düşünür
07Sinxronizasiya — bilik mərkəzə çatır
Nə quruldu Siyəzəndəki sensor bir dəyər ölçür. O dəyər validasiyadan keçir, dublikatı rədd olunur, anomaliyası ağıllı şəkildə qiymətləndirilir (səs-küy udulur, real problem alert yaradır), edge bazaya yazılır, sonra mərkəzi ERP-yə çatdırılır — və şəbəkə kəsilsə də itmir.

Bakıdakı rəhbər indi zavodun nəbzini görür. Kritik alert dəqiqələr içində ona çatır.

Bu, tam bir sənaye telemetriya sistemidir.
Buradan hara Dashboard — mərkəzdəki datanı vizuallaşdırmaq (Grafana, ya öz React panelin).
Real gateway — simulyatoru əsl Modbus/OPC-UA oxuyucu ilə əvəz etmək.
İkinci zavod — zavod_kod sütunu artıq hazırdır.
Proqnoz — toplanan data ilə nasazlığı əvvəlcədən görmək (predictive maintenance).

Amma bunların hamısı möhkəm bünövrə üzərində qurulur — və o bünövrə indi hazırdır.
